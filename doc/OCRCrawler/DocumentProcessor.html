def self.fetch(url, config = {})
  user_agent = (config && config[:user_agent]) || 'ruby-ocr-crawler/1.0'
  html = URI.parse(url).read('User-Agent' => user_agent)
  Nokogiri::HTML(html)
rescue StandardError => e
  Logger.warn("Failed to fetch #{url}: #{e.message}")
  nil
end
